Split text columns into 2 columns

dta <- tibble(text = c('HHM_AR', 'HMM_ID', 'HSMM_AR', 'HSMM_ID'))

dta
## # A tibble: 4 × 1
##   text   
##   <chr>  
## 1 HHM_AR 
## 2 HMM_ID 
## 3 HSMM_AR
## 4 HSMM_ID
dta %>%  
  separate(col = text, into = c('Hidden', 'Obs'), sep = '_' )
## # A tibble: 4 × 2
##   Hidden Obs  
##   <chr>  <chr>
## 1 HHM    AR   
## 2 HMM    ID   
## 3 HSMM   AR   
## 4 HSMM   ID
